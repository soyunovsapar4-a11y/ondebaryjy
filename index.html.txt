  <script>
    let currentUser = null;
    let analyses = [];
    let storageAvailable = true;

    // Проверяем доступность localStorage
    function isStorageAvailable() {
      try {
        localStorage.setItem('test', '1');
        localStorage.removeItem('test');
        return true;
      } catch (e) {
        storageAvailable = false;
        document.getElementById('storage-warning').classList.remove('hidden');
        return false;
      }
    }

    // Безопасное чтение из localStorage
    function loadAnalyses() {
      if (!storageAvailable) return [];
      try {
        return JSON.parse(localStorage.getItem('toprak_analyses')) || [];
      } catch (e) {
        return [];
      }
    }

    // Безопасная запись в localStorage
    function saveAnalyses(data) {
      if (!storageAvailable) return;
      try {
        localStorage.setItem('toprak_analyses', JSON.stringify(data));
      } catch (e) {}
    }

    const soilDB = {
      gumly: { name: "Gumly (песчаный)", plow: 18, waterFreq: "чаще", cropBase: "хлопок, дыни" },
      suymli: { name: "Çüýrli / Süýümli (суглинистый)", plow: 25, waterFreq: "средне", cropBase: "пшеница, ячмень" },
      killi: { name: "Killi (глинистый)", plow: 30, waterFreq: "реже", cropBase: "люцерна, овощи" }
    };

    function login() {
      const username = document.getElementById('login-username').value.trim() || "Daihan";
      currentUser = username;
      document.getElementById('username-display').textContent = username;
      document.getElementById('login-modal').classList.add('hidden');
      renderHistory();
    }

    function logout() {
      if (confirm('Выйти из аккаунта?')) {
        currentUser = null;
        document.getElementById('username-display').textContent = "Гость";
        document.getElementById('login-modal').classList.remove('hidden');
      }
    }

    function updateSliders() {
      document.getElementById('humus-val').textContent = document.getElementById('humus').value + '%';
      document.getElementById('soil-val').textContent = document.getElementById('soil-moisture').value + '%';
      document.getElementById('air-val').textContent = document.getElementById('air-humidity').value + '%';
    }

    function calculateAdvice() {
      const textureKey = document.getElementById('texture').value;
      const humus = parseFloat(document.getElementById('humus').value);
      const soilMoist = parseFloat(document.getElementById('soil-moisture').value);
      const airHum = parseFloat(document.getElementById('air-humidity').value);

      const tx = soilDB[textureKey];
      let plowDepth = tx.plow;
      if (humus < 1) plowDepth -= 3;
      if (soilMoist > 60) plowDepth += 2;

      let crop = tx.cropBase;
      if (humus > 2.5 && soilMoist > 40) crop = "Пшеница озимая / ячмень (богара)";
      else if (humus < 1) crop = "Хлопок + люцерна (севооборот)";

      let whenWater = airHum < 30 ? "Утром и вечером" : "Утром";
      let times = tx.waterFreq === "чаще" ? "5–7 раз" : tx.waterFreq === "средне" ? "3–5 раз" : "2–4 раза";
      if (soilMoist > 70) times = "Не поливать 7–10 дней";

      const html = `
        <div class="grid grid-cols-2 gap-6">
          <div class="bg-emerald-100 p-8 rounded-3xl"><p class="text-emerald-700 font-medium text-lg">Naçe sm çuňlukda sürmeli?</p><p class="text-5xl font-bold text-emerald-800 mt-2">${plowDepth} см</p></div>
          <div class="bg-emerald-100 p-8 rounded-3xl"><p class="text-emerald-700 font-medium text-lg">Näme ekin ekmeli?</p><p class="text-3xl font-bold text-emerald-800 mt-3">${crop}</p></div>
          <div class="bg-emerald-100 p-8 rounded-3xl"><p class="text-emerald-700 font-medium text-lg">Haçan suwarmaly?</p><p class="text-3xl font-bold text-emerald-800 mt-3">${whenWater}</p></div>
          <div class="bg-emerald-100 p-8 rounded-3xl"><p class="text-emerald-700 font-medium text-lg">Näçe gezek?</p><p class="text-3xl font-bold text-emerald-800 mt-3">${times}</p></div>
        </div>
        <p class="text-xs text-gray-500 mt-8">Рекомендации основаны на данных FAO и исследованиях почв Туркменистана (2023–2025).</p>
      `;

      document.getElementById('advice-content').innerHTML = html;
      document.getElementById('result').classList.remove('hidden');

      // Сохраняем анализ
      const analysis = {
        date: new Date().toLocaleString('ru-RU'),
        texture: tx.name,
        humus: humus,
        soilMoist: soilMoist,
        airHum: airHum,
        plow: plowDepth,
        crop: crop
      };
      analyses.unshift(analysis);
      if (analyses.length > 10) analyses.pop();
      saveAnalyses(analyses);
      renderHistory();
    }

    function renderHistory() {
      const container = document.getElementById('history-list');
      container.innerHTML = analyses.length === 0 
        ? `<p class="text-emerald-200 text-center py-8">Пока нет анализов</p>`
        : analyses.map((a, i) => `
          <div onclick="loadAnalysis(${i})" class="flex justify-between bg-white/10 hover:bg-white/20 p-4 rounded-2xl cursor-pointer">
            <div>
              <p class="font-medium">${a.date}</p>
              <p class="text-sm opacity-80">${a.texture} • ${a.humus}% гумуса</p>
            </div>
            <div class="text-right">
              <p class="text-emerald-400 text-2xl font-semibold">${a.plow} см</p>
              <p class="text-sm">${a.crop}</p>
            </div>
          </div>
        `).join('');
    }

    function loadAnalysis(i) {
      const a = analyses[i];
      alert(`📅 ${a.date}\n\nГлубина вспашки: ${a.plow} см\nРекомендуемая культура: ${a.crop}`);
    }

    function clearHistory() {
      if (confirm('Очистить всю историю?')) {
        analyses = [];
        saveAnalyses([]);
        renderHistory();
      }
    }

    function printReport() { window.print(); }

    function downloadPDF() {
      const { jsPDF } = window.jspdf;
      const doc = new jsPDF();
      doc.setFontSize(20);
      doc.text("Toprak Maslahatçy — Отчёт", 20, 25);
      doc.setFontSize(12);
      doc.text(`Дата: ${new Date().toLocaleString('ru-RU')}`, 20, 45);
      doc.text(document.getElementById('advice-content').innerText, 20, 65, { maxWidth: 170 });
      doc.save("toprak_maslahat_report.pdf");
      alert("✅ PDF сохранён!");
    }

    function syncCloud() {
      const msg = document.createElement('div');
      msg.style.cssText = 'position:fixed; bottom:30px; right:30px; background:#10b981; color:white; padding:20px 28px; border-radius:9999px; box-shadow:0 20px 25px -5px rgb(16 185 129); font-weight:600;';
      msg.textContent = '☁️ Синхронизация... Данные обновлены';
      document.body.appendChild(msg);
      setTimeout(() => msg.remove(), 2800);
    }

    // Инициализация
    window.onload = () => {
      isStorageAvailable();
      analyses = loadAnalyses();
      document.querySelectorAll('input[type="range"]').forEach(sl => sl.addEventListener('input', updateSliders));
      updateSliders();
      document.getElementById('login-modal').classList.remove('hidden');
      renderHistory();
    };
  </script>
